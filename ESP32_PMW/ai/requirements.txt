# PicoScope 5443D capture tool dependencies.
#
# IMPORTANT: the 5443D (5000D series) uses the ps5000a driver via the official
# `picosdk` Python wrapper. Do NOT install `pypicosdk` -- it only supports the
# 6000E/3000E series and will not open a 5443D.
#
# `picosdk` is a thin ctypes wrapper; it dlopen()s the NATIVE PicoSDK runtime,
# which must be installed SEPARATELY from https://www.picotech.com/downloads
# (it is not bundled in the pip package). Missing native runtime is the most
# common failure mode -- you'll see an OSError at `import picosdk.ps5000a`.
picosdk>=1.1

# Analysis / plotting (already present in writeup/.venv).
numpy>=1.24
scipy>=1.10
matplotlib>=3.7
